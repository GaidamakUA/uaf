var DEBUG_STRINGS = function() {
    this.m_debugStrings = {};

    this.AlreadyNoted = function(str)
    {
        var returnVal = false;
        if (this[str] != null) {
            this[string] = 1;
            returnVal = false;
        } else {
            returnVal = true;
        }
        return returnVal;
    }
}
